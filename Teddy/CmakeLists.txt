# shared library Teddy
add_library(Teddy SHARED
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/teddy_pch.h 
    ${CMAKE_CURRENT_SOURCE_DIR}/src/teddy_pch.cpp   
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Log.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Log.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/EntryPoint.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Core.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Application.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Application.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Events/Event.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Events/ApplicationEvent.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Events/KeyEvent.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Events/MouseEvent.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Events/SDL_Events.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Platform/OpenGL/OpenGlWindow.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Platform/OpenGL/OpenGlWindow.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/Teddy/Window.h 
    
)

target_precompile_headers(Teddy PRIVATE
                            src/teddy_pch.h)

target_include_directories(Teddy PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/src
)

target_compile_definitions(Teddy PRIVATE
    $<$<CONFIG:Debug>:TED_DEBUG>
    $<$<CONFIG:Release>:TED_RELEASE>
    $<$<CONFIG:Dist>:TED_DIST>
    TED_BUILD_DLL
)

set_target_properties(Teddy PROPERTIES
    LIBRARY_OUTPUT_DIRECTORY "${OUTPUT_DIR}/Teddy"
    RUNTIME_OUTPUT_DIRECTORY "${OUTPUT_DIR}/Teddy"
    PROJECT_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}"
    PCH_INCLUDE "teddy_pch.h"
)
