target_precompile_headers(Teddy PRIVATE
                            src/teddyPch.h)

target_include_directories(Teddy PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/src
)

target_compile_definitions(Teddy PRIVATE
    $<$<CONFIG:Debug>:TED_DEBUG>
    $<$<CONFIG:Release>:TED_RELEASE>
    $<$<CONFIG:Dist>:TED_DIST>
    TED_BUILD_DLL
    _CRT_SECURE_NO_WARNINGS
)

set_target_properties(Teddy PROPERTIES
    LIBRARY_OUTPUT_DIRECTORY "${OUTPUT_DIR}/Teddy"
    RUNTIME_OUTPUT_DIRECTORY "${OUTPUT_DIR}/Teddy"
    PROJECT_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}"
    PCH_INCLUDE "teddyPch.h"
)
